"""

SELECT ?person ?name
WHERE {
  ?person a rpc:Person .
  ?person rpc:name ?name .

#### LISTAR PUBS, TITLES E ANOS
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT ?pub ?title ?year
WHERE {
  ?pub a rpc:Publication .
  OPTIONAL { ?pub rpc:title ?title . }
  OPTIONAL { ?pub rpc:year ?year . }
}
LIMIT 50
"""

#### CONTAR O NÚMERO DE PUBLICAÇÕES POR TIPO
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT ?type (COUNT(?pub) AS ?count)
WHERE {
  ?pub a ?type .
  FILTER(?type IN (rpc:Publication, rpc:Article_in_Journal, rpc:Article_in_Proceedings, rpc:Book))
}
GROUP BY ?type
ORDER BY DESC(?count)
"""

#### LISTAR AUTORES, PUBS E ANOS
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT ?authorName ?pubTitle ?year
WHERE {
  ?person rpc:name ?authorName .
  ?person rpc:author_inv ?pub .
  OPTIONAL { ?pub rpc:title ?pubTitle . }
  OPTIONAL { ?pub rpc:year ?year . }
}
ORDER BY ?authorName
LIMIT 100
"""

#### AUTORES QUE PUBLICARAM LIVROS (BOOKS)
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT DISTINCT ?authorName ?bookTitle ?year
WHERE {
  ?person rpc:name ?authorName .
  ?person rpc:author_inv ?book .
  ?book a rpc:Book .
  OPTIONAL { ?book rpc:book_title ?bookTitle . }
  OPTIONAL { ?book rpc:year ?year . }
}
ORDER BY ?authorName
"""

#### AUTORES COM PUBLICAÇÕES EM JOURNALS (ARTICLE_IN_JOURNAL)
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT ?authorName ?journalName ?year
WHERE {
  ?person rpc:name ?authorName .
  ?person rpc:author_inv ?article .
  ?article a rpc:Article_in_Journal .
  OPTIONAL { ?article rpc:journal_name ?journalName . }
  OPTIONAL { ?article rpc:year ?year . }
}
ORDER BY ?authorName
LIMIT 100
"""

#### PUBLICAÇÕES MAIS RECENTES (ANO >= 2020)
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT ?title ?year ?type
WHERE {
  ?pub a ?type .
  ?pub rpc:title ?title .
  ?pub rpc:year ?year .
  FILTER(?type != rpc:Document)
  FILTER(xsd:integer(?year) >= 2020)
}
ORDER BY DESC(?year)
LIMIT 50
"""

#### PUBLICAÇÕES POR INTERVALO DE ANOS
"""
PREFIX rpc: <http://swat.cse.lehigh.edu/resources/onto/dblp.owl#>

SELECT (COUNT(?pub) AS ?count) ?year
WHERE {
  ?pub rpc:year ?year .
  FILTER(xsd:integer(?year) >= 1900 && xsd:integer(?year) <= 2026)
}
GROUP BY ?year
ORDER BY ?year
"""
